func
 
TestFilterOutDebugMethods
(
t
 
*
testing
.
T
)
 
{

	
t
.
Run
(
"OK"
,
 
func
(
t
 
*
testing
.
T
)
 
{

		
for
 
input
,
 
expectedOutput
 
:=
 
range
 
map
[
string
]
string
{

			
openAPIFile
:
 
openAPIFileWithoutDebugEndpoints
,

		
}
 
{

			
output
,
 
err
 
:=
 
httputils
.
FilterOutDebugMethods
(
input
)

			
helpers
.
FailOnError
(
t
,
 
err
)

			
assert
.
JSONEq
(
t
,
 
expectedOutput
,
 
output
)

		
}

	
}
)


	
t
.
Run
(
"Error"
,
 
func
(
t
 
*
testing
.
T
)
 
{

		
expectedErr
 
:=
 
`error unmarshaling JSON: Failed to unmarshal extension properties: `
 
+

			
`json: cannot unmarshal string into Go value of type map[string]json.RawMessage`
 
+
 
"\n"
 
+

			
`Input: "definitely-not-json"`

		
_
,
 
err
 
:=
 
httputils
.
FilterOutDebugMethods
(
"definitely-not-json"
)

		
assert
.
EqualError
(
t
,
 
err
,
 
expectedErr
)

	
}
)


}
