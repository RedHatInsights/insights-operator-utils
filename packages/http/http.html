ReplaceParamsInEndpointAndTrimLeftSlash replaces params in endpoint and trims left slash

 
func
 
ReplaceParamsInEndpointAndTrimLeftSlash
(
endpoint
,
 
replacer
 
string
)
 
string
 
{

	
re
 
:=

.
MustCompile
(
`\{[a-zA-Z_0-9]+\}`

.

(
endpoint
,
 
"/"
)

endpoint


}
