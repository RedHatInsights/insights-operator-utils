checkResponse checks status code against the expected one and check content-type + payload

(

t
.

)

	
}

!=

{

		
t
.

"Expected status code %v, got %v"

)

if

{

:=

.

.

if

{

.

)

body

:=
 
ioutil
.
ReadAll
(
res
.

)

		
if
 
err
 
!=
 
nil
 
{

			
t
.
Fatal
(
err
)

		
}

		
defer

(
t
,

)


		
fmt
.
Println
(

)


		
var

map
[
string
]
interface
{
}

.
NewDecoder

.

(

)
)
.

(
&
expected
)

		
if

!=
 
nil
 
{

			
t
.

(
"Response body is not JSON"
)

		
}


		
var

map
[
string
]
interface
{
}

.

)
